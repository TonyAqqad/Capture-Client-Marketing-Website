```typescript
export function generateHowToSchema(howto: {
  name: string;
  description: string;
  totalTime: string; // ISO 8601 duration format: PT48H
  steps: Array<{
    name: string;
    text: string;
    image?: string;
    url?: string;
  }>;
}) {
  return {
    "@context": "https://schema.org",
    "@type": "HowTo",
    name: howto.name,
    description: howto.description,
    totalTime: howto.totalTime,
    estimatedCost: {
      "@type": "MonetaryAmount",
      currency: "USD",
      value: "0" // Free consultation
    },
    tool: [
      {
        "@type": "HowToTool",
        name: "Capture Client Platform"
      }
    ],
    step: howto.steps.map((step, index) => ({
      "@type": "HowToStep",
      position: index + 1,
      name: step.name,
      text: step.text,
      image: step.image,
      url: step.url
    }))
  };
}
```
